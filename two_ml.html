import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer

# Загрузим данные из файла
file_path = 'no_vectors.xlsx'
df = pd.read_excel(file_path)

# Преобразуем столбец cleaned_text в формат пригодный для TfidfVectorizer
df['cleaned_text'] = df['cleaned_text'].apply(lambda x: ' '.join(eval(x)))

# Создаем объект TfidfVectorizer
tfidf_vectorizer = TfidfVectorizer()

# Преобразуем текстовые данные в TF-IDF векторы
tfidf_matrix = tfidf_vectorizer.fit_transform(df['cleaned_text'])
    
# Преобразуем матрицу TF-IDF в DataFrame
tfidf_df = pd.DataFrame(tfidf_matrix.toarray(), columns=tfidf_vectorizer.get_feature_names_out())

# Объединяем с оригинальным DataFrame
df = df.drop(columns=['cleaned_text'])
df_tfidf = pd.concat([df, tfidf_df], axis=1)
